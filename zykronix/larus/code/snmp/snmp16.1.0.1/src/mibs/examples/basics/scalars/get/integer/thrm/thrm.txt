   THERMOMETER-MIB DEFINITIONS ::= BEGIN

   IMPORTS
       MODULE-IDENTITY, OBJECT-TYPE
           FROM SNMPv2-SMI;

   snmpResearch                OBJECT IDENTIFIER ::= { enterprises 99 }

   snmpResearchMIBs            OBJECT IDENTIFIER ::= { snmpResearch 12 }

   srExamples                  OBJECT IDENTIFIER ::= { snmpResearchMIBs 19 }

   srBasics                    OBJECT IDENTIFIER ::= { srExamples 1 }

   srScalars                   OBJECT IDENTIFIER ::= { srBasics 1 }

   srScalarGet                 OBJECT IDENTIFIER ::= { srScalars 1 }

   srInteger                   OBJECT IDENTIFIER ::= { srScalarGet 1 }

   thermometer MODULE-IDENTITY
       LAST-UPDATED "9512130000Z"
       ORGANIZATION "SNMP Research"
       CONTACT-INFO
              "         Technical Support

                Postal: SNMP Research
                        3001 Kimberlin Heights Road
                        Knoxville, TN  37920-9716

                   Tel: +1 865 573 1434
                   Fax: +1 865 573 9197

                E-mail: support@snmp.com
                   WWW: http://www.snmp.com"
       DESCRIPTION
               "An example MIB module to demonstrate the SNMP
               Research get method routine and the internal
	       representation of INTEGER MIB objects."
       ::= { srInteger 1 }


   thrmTemperature OBJECT-TYPE
       SYNTAX  INTEGER (-460..2147483187)  -- -273 C = -459.4 F = 0 K
       MAX-ACCESS  read-only
       STATUS  current
       DESCRIPTION
               "The temperature reported by the thermometer."
       ::= { thermometer 1 }

   thrmUnits OBJECT-TYPE
       SYNTAX  INTEGER { celsius(1), fahrenheit(2) }
       MAX-ACCESS  read-only
       STATUS  current
       DESCRIPTION
               "The unit of measurement for the temperature reported
	       by the thermometer."
       ::= { thermometer 2 }

END
